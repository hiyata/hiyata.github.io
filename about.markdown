<script>
document.addEventListener('DOMContentLoaded', (event) => {
  // DNA Animation
  const svg = document.getElementById('dna-svg');
  const width = 1000;
  const height = 200;
  const numBases = 20;
  const baseWidth = width / numBases;
  const helixOffset = 40;

  for (let i = 0; i < numBases; i++) {
    const x = i * baseWidth + baseWidth / 2;
    const y1 = height / 2 + helixOffset * Math.sin(i * 2 * Math.PI / numBases + Math.PI);
    const y2 = height / 2 + helixOffset * Math.sin(i * 2 * Math.PI / numBases);

    const circle1 = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
    circle1.setAttribute('cx', x);
    circle1.setAttribute('cy', y1);
    circle1.setAttribute('r', '5');
    circle1.setAttribute('fill', '#667eea');
    svg.appendChild(circle1);

    const circle2 = document.createElementNS('http://www.w3.org/2000/svg', 'circle');
    circle2.setAttribute('cx', x);
    circle2.setAttribute('cy', y2);
    circle2.setAttribute('r', '5');
    circle2.setAttribute('fill', '#764ba2');
    svg.appendChild(circle2);
  }

  anime({
    targets: '#dna-svg circle:nth-child(odd)',
    translateY: function(el) {
      return parseFloat(el.nextElementSibling.getAttribute('cy')) - parseFloat(el.getAttribute('cy'));
    },
    duration: 1000,
    loop: true,
    easing: 'linear',
    delay: function(el, i) {
      return i * 100;
    },
    direction: 'alternate'
  });

  anime({
    targets: '#dna-svg circle:nth-child(even)',
    translateY: function(el) {
      return parseFloat(el.previousElementSibling.getAttribute('cy')) - parseFloat(el.getAttribute('cy'));
    },
    duration: 1000,
    loop: true,
    easing: 'linear',
    delay: function(el, i) {
      return 50 + i * 100;
    },
    direction: 'alternate'
  });

  // Scroll animations
  const animateOnScroll = (targets, translateY = 50) => {
    anime({
      targets: targets,
      opacity: [0, 1],
      translateY: [translateY, 0],
      duration: 1000,
      easing: 'easeOutQuad',
      delay: anime.stagger(100),
      begin: function(anim) {
        document.querySelectorAll(targets).forEach(el => {
          el.style.opacity = '0';
          el.style.transform = `translateY(${translateY}px)`;
        });
      }
    });
  };

  const observerOptions = {
    threshold: 0.1
  };

  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        if (entry.target.classList.contains('intro-section')) {
          animateOnScroll('.intro-text, .intro-vision');
        } else if (entry.target.classList.contains('skills-section')) {
          animateOnScroll('.skill-category');
        } else if (entry.target.classList.contains('personal-interests')) {
          animateOnScroll('.interest-item');
        } else if (entry.target.classList.contains('publications')) {
          animateOnScroll('.publication-item', 20);
        }
        observer.unobserve(entry.target);
      }
    });
  }, observerOptions);

  document.querySelectorAll('.intro-section, .skills-section, .personal-interests, .publications').forEach(section => {
    observer.observe(section);
  });

  // Hero section animations
  anime({
    targets: '.profile-image-container',
    scale: [0, 1],
    opacity: [0, 1],
    duration: 1000,
    easing: 'easeOutElastic(1, .8)'
  });

  anime({
    targets: ['.name', '.title'],
    opacity: [0, 1],
    translateY: [50, 0],
    duration: 1000,
    delay: anime.stagger(200),
    easing: 'easeOutQuad'
  });

  anime({
    targets: '.social-icon',
    opacity: [0, 1],
    translateY: [20, 0],
    duration: 800,
    delay: anime.stagger(100),
    easing: 'easeOutQuad'
  });

  // Skill item hover effect
  const skillItems = document.querySelectorAll('.skill-item');
  skillItems.forEach(item => {
    item.addEventListener('mouseenter', () => {
      anime({
        targets: item,
        scale: 1.05,
        boxShadow: '0 8px 20px rgba(0, 0, 0, 0.2)',
        duration: 300,
        easing: 'easeOutQuad'
      });
    });
    item.addEventListener('mouseleave', () => {
      anime({
        targets: item,
        scale: 1,
        boxShadow: '0 5px 15px rgba(0, 0, 0, 0.1)',
        duration: 300,
        easing: 'easeOutQuad'
      });
    });
  });
});
</script>
